Our mass-loss prescription joins together three different scalings.  The basic main-sequence scaling is
%
\begin{equation}
  \begin{split}
    \log\!\left(\frac{\dot{M}_{\rm MS}}{M_\odot\ \mathrm{yr}^{-1}}\right)
      &= -6 + 1.5\log\!\left(\frac{L_{\rm MS}}{10^{6}\,L_\odot}\right) \\
      &\quad + 0.69\log\!\left(\frac{Z}{Z_\odot}\right) ,
  \end{split}
  \label{eq:mdot_MS}
\end{equation}
%
the weak-wind cap is
%
\begin{equation}
  \begin{split}
    \log\!\left(\frac{\dot{M}_{\rm weak}}{M_\odot\ \mathrm{yr}^{-1}}\right)
      &=  2.9\log\!\left(\frac{L_{\rm MS}}{10^{5}\,L_\odot}\right) -7.65 ,
  \end{split}
  \label{eq:mdot_weak}
\end{equation}
%
and the rate for very-massive stars is \citep{sabhahit:2022.vms.mass.loss} is
%
\begin{equation}
  \begin{split}
    \log\!\left(\frac{\dot{M}_{\rm S22}}{M_\odot\ \mathrm{yr}^{-1}}\right)
      &= -8.445 + 4.77\log\!\frac{L_{\rm MS}}{10^5\,L_\odot} \\
      &\quad - 3.99\log\!\frac{M_*}{30\,M_\odot}
             - 1.226\log\!\frac{v_\infty/v_{\rm esc}}{2} \\
      &\quad + 0.761\log\!\frac{Z}{Z_\odot} ,
  \end{split}
  \label{eq:mdot_S22}
\end{equation}

where for $v_\infty$ we use \citet{lamers_1995_wind_vesc}:
%
\begin{equation}
  \frac{v_\infty}{v_{\rm esc}} =
  \begin{cases}
    0.7 & T_{\rm eff} < 12\,500\,\mathrm{K},\\
    1.3 & 12\,500\,\mathrm{K} \le T_{\rm eff} < 21\,000\,\mathrm{K},\\
    2.6 & T_{\rm eff} \ge 21\,000\,\mathrm{K},
  \end{cases}
  \label{eq:vinf_vesc}
\end{equation}
%
and $v_{\rm esc} = \sqrt{2GM_*/R_*}$.
%
The final wind mass-loss rate used is then:
%
\begin{equation}
  \dot{M}_{\rm wind} =
    \max\!\Bigl(\min\!\bigl(\dot{M}_{\rm MS},\,\dot{M}_{\rm weak}\bigr),\;
                \dot{M}_{\rm S22}\Bigr),
  \label{eq:mdot_wind}
\end{equation}
which we plot as a function of mass and metallicity in Figure~\ref{fig:wind_mdot}.
